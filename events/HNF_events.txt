#################################################
###     Slovakia
#################################################

# Hungary asks for Slovakia
country_event = {
	id = hf.1
	title = hf.1.t
	desc = hf.1.d
	picture = GFX_report_event_vienna_award_hungary
	
	is_triggered_only = yes
	
	# Give Hungary Slovakia
	option = {
		trigger = {
			NOT = { has_dlc = "Death or Dishonor" }
			OR = {
				GER = {
					controls_state = 70
					controls_state = 71
				}
				SLO = {
					controls_state = 70
					controls_state = 71
				}
				CZE = {
					controls_state = 70
					controls_state = 71
				}
			}
		}
		name = hf.1.a
		ai_chance = { 
			factor = 80
			modifier = {
				factor = 0.5
				is_historical_focus_on = yes
			}
		}
		HUN = {
			country_event = hf.2
		}	
	}
	
	# Reject Hungarian request
	option = { 
		name = hf.1.b
		ai_chance = { 
			factor = 10
		}
		### add_named_threat = { threat = 3 name = GER_second_vienna_award_HUN_all }
		HUN = { 
			country_event = hf.3
		}
	}
}

# Slovakia given to Hungary
country_event = {
	id = hf.2
	title = hf.2.t
	desc = hf.2.d
	picture = GFX_report_event_vienna_award_hungary
	
	is_triggered_only = yes
	
	# Give Hungary Slovakia
	option = { 
		name = hf.2.a
		ai_chance = { 
			factor = 10
		}
		HUN = {
			annex_country = { target = SLO transfer_troops = no }
		}
	}
}

# Hungary rejected for Slovakia
country_event = {
	id = hf.3
	title = hf.3.t
	desc = hf.3.d
	picture = GFX_report_event_vienna_award_hungary
	
	is_triggered_only = yes
	
	# Give Hungary Slovakia - Germany owns Slovakia
	option = { 
		trigger = {
			GER = {
				controls_state = 70
				controls_state = 71
			}
		}
		name = hf.3.a
		ai_chance = { 
			factor = 10
		}
		create_wargoal = {
			type = take_claimed_state
			target = GER
		}
	}
	# Give Hungary Slovakia - Slovakia exists
	option = { 
		trigger = {
			SLO = {
				controls_state = 70
				controls_state = 71
			}
		}
		name = hf.3.b
		ai_chance = { 
			factor = 10
		}
		create_wargoal = {
			type = annex_everything
			target = SLO
		}
	}
}
